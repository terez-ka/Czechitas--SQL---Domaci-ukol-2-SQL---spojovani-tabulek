--Zadání domácího úkolu:

--Ve kterém segmentu je nejvíce produktù?
SELECT count (distinct productid) AS celkovy_pocet_produktu, segment
FROM SQL1.Products
Group by segment
Order by celkovy_pocet_produktu desc
FETCH FIRST 1 ROW ONLY;

--Vytvoø pøehled jaké segmenty jsou zastoupeny v jednotlivých kategoriích, kolik výrobkù obsahují a jaká je prùmìrná cena výrobku v tomto èlenìní.
SELECT count (distinct productid) AS celkovy_pocet_vyrobku, ROUND ((AVG (pricenew)),2) AS prumerna_cena_vyrobku, category, segment
FROM SQL1.Products
Group by category, segment
Order by category asc;


--Kterých výrobkù (ProductID) se prodalo nejvíce v dubnu 2015? (zajímají nás kusy - units)
SELECT SUM (UNITS) as pocet_prodanych_kusu, productid
FROM SQL1.Sales
WHERE to_char (calendarid, 'MM.YYYY') = '04.2015'
Group by productid
Order by pocet_prodanych_kusu desc
FETCH FIRST 1 ROWS ONLY;
